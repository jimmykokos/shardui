-- Утилиты для безопасного доступа
local function SafeWaitChild(parent, name, timeout)
    timeout = timeout or 5
    local startTime = tick()
    while tick() - startTime < timeout do
        local child = parent:FindFirstChild(name)
        if child then return child end
        RunService.Heartbeat:Wait()
    end
    return nil
end

local function SafeFindFirstChild(parent, name)
    local success, result = pcall(function()
        return parent:FindFirstChild(name)
    end)
    return success and result or nil
end

local function SafeDestroy(obj)
    if obj and obj.Parent then
        pcall(function() obj:Destroy() end)
    end
end

local function SafeProperty(obj, prop)
    local success, result = pcall(function()
        return obj[prop]
    end)
    return success and result or nil
end

local function SafeSetProperty(obj, prop, value)
    local success = pcall(function()
        obj[prop] = value
    end)
    return success
end

local function SafeConnect(signal, callback)
    local success, connection = pcall(function()
        return signal:Connect(callback)
    end)
    return success and connection or nil
end

local function SafeDisconnect(connection)
    if connection and connection.Disconnect then
        pcall(function() connection:Disconnect() end)
    end
end

local function GetCharacter()
    return LocalPlayer.Character
end

local function GetHumanoid()
    local char = GetCharacter()
    if char then
        return SafeFindFirstChild(char, "Humanoid")
    end
    return nil
end

local function GetHRP()
    local char = GetCharacter()
    if char then
        return SafeFindFirstChild(char, "HumanoidRootPart")
    end
    return nil
end

local function GetAllPlayers()
    local list = {}
    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= LocalPlayer then
            table.insert(list, plr)
        end
    end
    return list
end

local function GetDistance(pos1, pos2)
    if not pos1 or not pos2 then return math.huge end
    return (pos1 - pos2).Magnitude
end

local function GetClosestPlayer(range)
    range = range or math.huge
    local closest = nil
    local minDist = range
    local localHRP = GetHRP()
    if not localHRP then return nil end
    
    for _, plr in ipairs(GetAllPlayers()) do
        local char = plr.Character
        if char then
            local hrp = SafeFindFirstChild(char, "HumanoidRootPart")
            if hrp then
                local dist = GetDistance(localHRP.Position, hrp.Position)
                if dist < minDist then
                    minDist = dist
                    closest = plr
                end
            end
        end
    end
    return closest
end

local function SafeNotify(title, text, duration)
    duration = duration or 3
    pcall(function()
        StarterGui:SetCore("SendNotification", {
            Title = title,
            Text = text,
            Duration = duration
        })
    end)
end

local function CopyToClipboard(text)
    pcall(function()
        if setclipboard then
            setclipboard(text)
        end
    end)
end

local ModuleManager = {}
ModuleManager.modules = {}

    "Invis", "Move", "Player", "Combat",

function ModuleManager:Register(name, category, config)
    if not name or not category then return end
    
    config = config or {}
    local module = {
        name = name,
        category = category,
        desc = config.desc or "",
        modType = config.modType or "toggle",
        enabled = false,
        settings = config.settings or {},
        values = {},
        connections = {},
        objects = {},
        _on = config.on or function() end,
        _off = config.off or function() end,
        _onSettingChange = config.onSettingChange or function() end,
        uiElements = {}
    }
    
    for _, setting in ipairs(module.settings) do
        module.values[setting.key] = setting.default
    end
    
    ModuleManager.modules[name] = module
end

function ModuleManager:Get(name)
    return ModuleManager.modules[name]
end

function ModuleManager:GetByCategory(category)
    local list = {}
    for name, mod in pairs(ModuleManager.modules) do
        if mod.category == category then
            table.insert(list, mod)
        end
    end
    return list
end

function ModuleManager:IsEnabled(name)
    local mod = ModuleManager.modules[name]
    return mod and mod.enabled or false
end

function ModuleManager:SetSetting(name, key, value)
    local mod = ModuleManager.modules[name]
    if mod and mod.values[key] ~= nil then
        mod.values[key] = value
        local success, err = pcall(function()
            mod._onSettingChange(key, value)
        end)
        if not success then
            warn("Setting change error: " .. tostring(err))
        end
    end
end

function ModuleManager:GetSetting(name, key)
    local mod = ModuleManager.modules[name]
    if mod then
        return mod.values[key]
    end
    return nil
end

function ModuleManager:Toggle(name)
    local mod = ModuleManager.modules[name]
    if not mod then return end
    
    if mod.modType == "button" then
        local success, err = pcall(function()
            mod._on()
        end)
        if not success then
            warn("Button error: " .. tostring(err))
        end
        return
    end
    
    mod.enabled = not mod.enabled
    
    if mod.enabled then
        local success, err = pcall(function()
            mod._on()
        end)
        if not success then
            warn("Toggle on error: " .. tostring(err))
            mod.enabled = false
        end
    else
        local success, err = pcall(function()
            mod._off()
        end)
        if not success then
            warn("Toggle off error: " .. tostring(err))
        end
        for _, conn in ipairs(mod.connections) do
            SafeDisconnect(conn)
        end
        mod.connections = {}
        for _, obj in ipairs(mod.objects) do
            SafeDestroy(obj)
        end
        mod.objects = {}
    end
    
    ModuleManager:UpdateModuleUI(name)
end

function ModuleManager:UpdateModuleUI(name)
    local mod = ModuleManager.modules[name]
    if not mod or not mod.uiElements.toggle then return end
    
    local toggle = mod.uiElements.toggle
    local indicator = toggle:FindFirstChild("Indicator")
    
    if indicator then
        local targetColor = mod.enabled and Color3.fromHex("#28DC64") or Color3.fromHex("#FF3741")
        local targetPos = mod.enabled and UDim2.new(1, -18, 0.5, -7) or UDim2.new(0, 2, 0.5, -7)
        
        TweenService:Create(indicator, TweenInfo.new(0.25, Enum.EasingStyle.Quad), {
            BackgroundColor3 = targetColor,
            Position = targetPos
        }):Play()
    end
end

function ModuleManager:Enable(name)
    local mod = ModuleManager.modules[name]
    if mod and not mod.enabled then
        self:Toggle(name)
    end
end

function ModuleManager:Disable(name)
    local mod = ModuleManager.modules[name]
    if mod and mod.enabled then
        self:Toggle(name)
    end
end

function ModuleManager:DisableAll()
    for name, mod in pairs(self.modules) do
        if mod.enabled then
            self:Toggle(name)
        end
    end
end

function ModuleManager:GetEnabledCount()
    local count = 0
    for _, mod in pairs(self.modules) do
        if mod.enabled then count = count + 1 end
    end
    return count
end

    container.BackgroundTransparency = 1
    container.Parent = parent

    label.Name = "Label"

    label.TextColor3 = Colors.Text
    label.TextSize = 12
    label.Font = Enum.Font.Gotham
    label.TextXAlignment = Enum.TextXAlignment.Left
    label.Parent = container
    
    local valueLabel = Instance.new("TextLabel")
    valueLabel.Name = "Value"
    valueLabel.Size = UDim2.new(0.5, 0, 0, 20)
    valueLabel.Position = UDim2.new(0.5, 0, 0, 0)
    valueLabel.BackgroundTransparency = 1
    valueLabel.Text = tostring(setting.default)
    valueLabel.TextColor3 = Colors.Accent
    valueLabel.TextSize = 12
    valueLabel.Font = Enum.Font.GothamBold
    valueLabel.TextXAlignment = Enum.TextXAlignment.Right
    valueLabel.Parent = container
    
    local track = Instance.new("Frame")
    track.Name = "Track"
    track.Size = UDim2.new(1, 0, 0, 6)
    track.Position = UDim2.new(0, 0, 0, 30)
    track.BackgroundColor3 = Colors.CardHover
    track.BorderSizePixel = 0
    track.Parent = container
    
    local trackCorner = Instance.new("UICorner")
    trackCorner.CornerRadius = UDim.new(1, 0)
    trackCorner.Parent = track
    
    local fill = Instance.new("Frame")
    fill.Name = "Fill"
    fill.Size = UDim2.new(0.5, 0, 1, 0)
    fill.BackgroundColor3 = Colors.Accent
    fill.BorderSizePixel = 0
    fill.Parent = track
    
    local fillCorner = Instance.new("UICorner")
    fillCorner.CornerRadius = UDim.new(1, 0)
    fillCorner.Parent = fill
    
    local knob = Instance.new("TextButton")
    knob.Name = "Knob"
    knob.Size = UDim2.new(0, 14, 0, 14)
    knob.Position = UDim2.new(0.5, -7, 0.5, -7)
    knob.BackgroundColor3 = Colors.Text
    knob.BorderSizePixel = 0
    knob.Text = ""
    knob.Parent = track
    
    local knobCorner = Instance.new("UICorner")
    knobCorner.CornerRadius = UDim.new(1, 0)
    knobCorner.Parent = knob
    
    local minVal = setting.min or 0
    local maxVal = setting.max or 100
    local currentVal = setting.default or minVal
    
    local function updateSlider(val)
        currentVal = math.clamp(val, minVal, maxVal)
        local percent = (currentVal - minVal) / (maxVal - minVal)
        fill.Size = UDim2.new(percent, 0, 1, 0)
        knob.Position = UDim2.new(percent, -7, 0.5, -7)
        valueLabel.Text = string.format("%.2f", currentVal)
        if onChange then onChange(currentVal) end
    end
    
    local dragging = false
    
    knob.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = true
        end
    end)
    
    track.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            local pos = input.Position.X - track.AbsolutePosition.X
            local percent = math.clamp(pos / track.AbsoluteSize.X, 0, 1)
            updateSlider(minVal + (maxVal - minVal) * percent)
        end
    end)
    
    UserInputService.InputChanged:Connect(function(input)
        if dragging and input.UserInputType == Enum.UserInputType.MouseMovement then
            local pos = input.Position.X - track.AbsolutePosition.X
            local percent = math.clamp(pos / track.AbsoluteSize.X, 0, 1)
            updateSlider(minVal + (maxVal - minVal) * percent)
        end
    end)
    
    UserInputService.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 then
            dragging = false
        end
    end)
    
    updateSlider(currentVal)
    
    return container, updateSlider
end

    local selected = dropdown.Text
    
    dropdown.MouseButton1Click:Connect(function()
        for i, opt in ipairs(options) do
            if opt == selected then
                local nextIdx = (i % #options) + 1
                selected = options[nextIdx]
                dropdown.Text = selected
                if onChange then onChange(selected) end
                break
            end
        end
    end)
    
    return container
end

function CreateSettingsPanel(parent, module)
    local panel = Instance.new("Frame")
    panel.Name = "SettingsPanel"
    panel.Size = UDim2.new(1, 0, 0, 0)
    panel.Position = UDim2.new(0, 0, 1, 5)
    panel.BackgroundColor3 = Colors.Sidebar
    panel.BorderSizePixel = 0
    panel.ClipsDescendants = true
    panel.Visible = false
    panel.Parent = parent
    
    local corner = Instance.new("UICorner")
    corner.CornerRadius = UDim.new(0, 6)
    corner.Parent = panel
    
    local layout = Instance.new("UIListLayout")
    layout.Padding = UDim.new(0, 8)
    layout.Parent = panel
    
    local padding = Instance.new("UIPadding")
    padding.PaddingLeft = UDim.new(0, 10)
    padding.PaddingRight = UDim.new(0, 10)
    padding.PaddingTop = UDim.new(0, 10)
    padding.PaddingBottom = UDim.new(0, 10)
    padding.Parent = panel
    
    local settingControls = {}
    
    for _, setting in ipairs(module.settings) do
        local control = nil
        
        if setting.type == "slider" then
            control = CreateSlider(panel, setting, function(val)
                ModuleManager:SetSetting(module.name, setting.key, val)
            end)
        elseif setting.type == "toggle" then
            local row = Instance.new("Frame")
            row.Size = UDim2.new(1, 0, 0, 25)
            row.BackgroundTransparency = 1
            row.Parent = panel
            
            local lbl = Instance.new("TextLabel")
            lbl.Size = UDim2.new(0.7, 0, 1, 0)
            lbl.BackgroundTransparency = 1
            lbl.Text = setting.label
            lbl.TextColor3 = Colors.Text
            lbl.TextSize = 12
            lbl.Font = Enum.Font.Gotham
            lbl.TextXAlignment = Enum.TextXAlignment.Left
            lbl.Parent = row
            
            local toggle, _ = CreateToggleSwitch(row, UDim2.new(1, -40, 0.5, -9), nil, function()
                local newVal = not module.values[setting.key]
                ModuleManager:SetSetting(module.name, setting.key, newVal)
            end)
            
            control = row
        elseif setting.type == "dropdown" then
            control = CreateDropdown(panel, setting, function(val)
                ModuleManager:SetSetting(module.name, setting.key, val)
            end)
        elseif setting.type == "textbox" then
            control = CreateTextbox(panel, setting, function(val)
                ModuleManager:SetSetting(module.name, setting.key, val)
            end)
        elseif setting.type == "color" then
            control = CreateColorPicker(panel, setting, function(val)
                ModuleManager:SetSetting(module.name, setting.key, val)
            end)
        end
        
        if control then
            settingControls[setting.key] = control
        end
    end
    
    local totalHeight = 20
    for _, setting in ipairs(module.settings) do
        if setting.type == "slider" then
            totalHeight = totalHeight + 53
        elseif setting.type == "color" then
            totalHeight = totalHeight + 85
        else
            totalHeight = totalHeight + 33
        end
    end
    
    panel:SetAttribute("targetHeight", totalHeight)
    
    return panel
end

function UpdateCategorySelection(buttons)
    for catName, elements in pairs(buttons) do
        local isActive = catName == State.currentCategory
        elements.indicator.Visible = isActive
        
        if isActive then
            TweenService:Create(elements.button, TweenInfo.new(0.15), {
                BackgroundColor3 = Colors.CardHover,
                TextColor3 = Colors.Accent
            }):Play()
        else
            TweenService:Create(elements.button, TweenInfo.new(0.15), {
                BackgroundColor3 = Colors.Card,
                TextColor3 = Colors.Text
            }):Play()
        end
    end
end

function UpdateContentArea()
    local content = ShardUI.elements.contentArea
    if not content then return end
    
    for _, child in ipairs(content:GetChildren()) do
        if child:IsA("Frame") then
            child:Destroy()
        end
    end
    
    local modules = ModuleManager:GetByCategory(State.currentCategory)
    
    for _, module in ipairs(modules) do
        CreateModuleRow(content, module)
    end
    
    task.delay(0.1, function()
        local totalHeight = 20
        for _, child in ipairs(content:GetChildren()) do
            if child:IsA("Frame") then
                totalHeight = totalHeight + child.Size.Y.Offset + 8
            end
        end
        content.CanvasSize = UDim2.new(0, 0, 0, totalHeight)
    end)
end

function UpdateHUD()
    if ShardUI.hudElements.watermarkText then
        local enabledCount = ModuleManager:GetEnabledCount()
        ShardUI.hudElements.watermarkText.Text = string.format(
            "SHARD DLC v5 | %d FPS | %d features | %dms",
            State.fps,
            enabledCount,
            State.ping
        )
    end
    
    if ShardUI.hudElements.speedText then
        local hrp = GetHRP()
        if hrp then
            local velocity = SafeProperty(hrp, "Velocity")
            if velocity then
                local speed = math.floor(velocity.Magnitude * 3.6)
                local color = Colors.Accent
                if speed > 100 then
                    color = Colors.Red
                elseif speed > 50 then
                    color = Color3.fromRGB(255, 220, 0)
                end
                ShardUI.hudElements.speedText.Text = speed .. " km/h"
                ShardUI.hudElements.speedText.TextColor3 = color
            end
        end
    end
    
    if ShardUI.hudElements.minimapDots then
        for _, child in ipairs(ShardUI.hudElements.minimapDots:GetChildren()) do
            child:Destroy()
        end
        
        local localHRP = GetHRP()
        if localHRP then
            local localPos = localHRP.Position
            local scale = 200 / 150
            
            for _, plr in ipairs(GetAllPlayers()) do
                local char = plr.Character
                if char then
                    local hrp = SafeFindFirstChild(char, "HumanoidRootPart")
                    if hrp then
                        local pos = hrp.Position
                        local relX = (pos.X - localPos.X) / scale
                        local relZ = (pos.Z - localPos.Z) / scale
                        
                        if math.abs(relX) < 70 and math.abs(relZ) < 70 then
                            local dot = Instance.new("Frame")
                            dot.Size = UDim2.new(0, 6, 0, 6)
                            dot.Position = UDim2.new(0.5, relX - 3, 0.5, relZ - 3)
                            dot.BackgroundColor3 = Colors.Red
                            dot.BorderSizePixel = 0
                            dot.Parent = ShardUI.hudElements.minimapDots
                            
                            local dotCorner = Instance.new("UICorner")
                            dotCorner.CornerRadius = UDim.new(1, 0)
                            dotCorner.Parent = dot
                        end
                    end
                end
            end
        end
    end
    
    if ShardUI.hudElements.targetHUD then
        local closest = GetClosestPlayer(100)
        if closest then
            local char = closest.Character
            if char then
                local hrp = SafeFindFirstChild(char, "HumanoidRootPart")
                local humanoid = SafeFindFirstChild(char, "Humanoid")
                local localHRP = GetHRP()
                
                if hrp and humanoid and localHRP then
                    ShardUI.hudElements.targetHUD.Visible = true
                    ShardUI.hudElements.targetName.Text = closest.DisplayName or closest.Name
                    local dist = GetDistance(localHRP.Position, hrp.Position)
                    ShardUI.hudElements.targetDist.Text = math.floor(dist) .. " studs"
                    
                    local hp = SafeProperty(humanoid, "Health") or 0
                    local maxHp = SafeProperty(humanoid, "MaxHealth") or 100
                    local hpPercent = math.clamp(hp / maxHp, 0, 1)
                    ShardUI.hudElements.targetHP.Size = UDim2.new(hpPercent, 0, 1, 0)
                end
            end
        else
            ShardUI.hudElements.targetHUD.Visible = false
        end
    end
    
    if ShardUI.hudElements.keybindDisplay then
        for _, child in ipairs(ShardUI.hudElements.keybindDisplay:GetChildren()) do
            if child:IsA("TextLabel") then
                child:Destroy()
            end
        end
        
        local activeCount = 0
        for key, moduleName in pairs(State.keybinds) do
            local mod = ModuleManager:Get(moduleName)
            if mod and mod.enabled then
                activeCount = activeCount + 1
                local bindLabel = Instance.new("TextLabel")
                bindLabel.Size = UDim2.new(1, 0, 0, 16)
                bindLabel.BackgroundTransparency = 1
                bindLabel.Text = "[" .. tostring(key) .. "] " .. moduleName
                bindLabel.TextColor3 = Colors.Accent
                bindLabel.TextSize = 10
                bindLabel.Font = Enum.Font.Gotham
                bindLabel.TextXAlignment = Enum.TextXAlignment.Left
                bindLabel.Parent = ShardUI.hudElements.keybindDisplay
            end
        end
        
        ShardUI.hudElements.keybindDisplay.Size = UDim2.new(0, 150, 0, math.max(30, activeCount * 20 + 16))
    end
end

                    end)
